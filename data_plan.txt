users (table users)
    user_id
    first_name VARCHAR
    last_name VARCHAR
    passwordhash VARCHAR
    email VARCHAR
    address VARCHAR

order_history (new table order_history)
    order_id INT
    item_id (relate to items.id) INT
    quantity INT
    order date DATE
    user_id (realate to user.id) INT

cart (new table cart)
    user id (relate to users) INT
    item id (relate to item) INT
    quantity INT

items (table items)
    id INT
    name VARCHAR
    description VARCHAR
    price MONEY
    make VARCHAR
